```typescript
import { Context } from "@acme/edge-functions";

export default async (request: Request, context: Context) => {
  const url = new URL(request.url);
  
  // Check for existing variant cookie
  const existingVariant = context.cookies.get("ab_variant");
  if (existingVariant) {
    url.pathname = url.pathname.replace("/home", `/home-${existingVariant}`);
    return context.next({ request: new Request(url, request) });
  }
  
  // Assign variant
  const variant = Math.random() < 0.5 ? "a" : "b";
  const response = await context.next({
    request: new Request(
      url.pathname.replace("/home", `/home-${variant}`),
      request
    )
  });
  
  response.headers.append("Set-Cookie", `ab_variant=${variant}; Path=/; Max-Age=86400`);
  return response;
};
```
